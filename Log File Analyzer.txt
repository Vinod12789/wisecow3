LOG_FILE="/var/log/apache2/access.log"  # Update with your actual log file path

# Count 404 errors
echo "Number of 404 Errors: $(awk '($9 ~ /404/)' $LOG_FILE | wc -l)"

# Most requested pages
echo "Most Requested Pages:"
awk '{print $7}' $LOG_FILE | sort | uniq -c | sort -nr | head -n 10

# IP addresses with the most requests
echo "IP Addresses with the Most Requests:"
awk '{print $1}' $LOG_FILE | sort | uniq -c | sort -nr | head -n 10
